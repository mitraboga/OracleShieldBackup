# OracleShieldBackup cron schedule
# Install as the oracle user after editing PROJECT_ROOT/config/backup.conf:
#   crontab /opt/oracle-shield-backup/cron/crontab.txt

SHELL=/bin/bash
PATH=/usr/local/bin:/usr/bin:/bin
BACKUP_CONFIG=/opt/oracle-shield-backup/config/backup.conf

# Daily cumulative incremental backup at 01:00.
0 1 * * 1-6 /opt/oracle-shield-backup/scripts/incremental_backup.sh

# Weekly level 0 full backup on Sunday at 01:00.
0 1 * * 0 /opt/oracle-shield-backup/scripts/full_backup.sh

# Archive log backup every 4 hours.
0 */4 * * * /opt/oracle-shield-backup/scripts/archive_backup.sh

# Recovery readiness validation daily at 05:00.
0 5 * * * /opt/oracle-shield-backup/scripts/validate_backup.sh

# Local retention cleanup and dashboard refresh.
30 5 * * * /opt/oracle-shield-backup/scripts/cleanup_old_backups.sh
45 5 * * * /opt/oracle-shield-backup/scripts/generate_report.sh
